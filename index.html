        function extract(prompt, type) {
            const patterns = {
                headline: /headline[:\s]+["']?([^"'\n,]+)/i,
                subheadline: /subheadline[:\s]+["']?([^"'\n,]+)/i,
                cta: /cta|button[:\s]+["']?([^"'\n,]+)/i
            };
            const match = prompt.match(patterns[type]);
            return match ? match[1].trim() : null;
        }

        function parsePrompt(prompt) {
            const lower = prompt.toLowerCase();
            if (lower.includes('hero') || lower.includes('landing')) {
                return TEMPLATES.hero(
                    extract(prompt, 'title') || 'Build Faster',
                    extract(prompt, 'subtitle') || 'Describe what you want. Get working code.',
                    extract(prompt, 'cta') || 'Get Started'
                );
            } else if (lower.includes('pricing') || lower.includes('plan') || lower.includes('cost')) {
                return TEMPLATES.pricing(
                    extract(prompt, 'title') || 'Simple Pricing',
                    extract(prompt, 'subtitle') || 'Choose the plan that fits your needs'
                );
            } else if (lower.includes('feature') || lower.includes('benefit')) {
                return TEMPLATES.feature(
                    extract(prompt, 'title') || 'Why Choose Us',
                    extract(prompt, 'subtitle') || 'Everything you need to succeed'
                );
            }
            return TEMPLATES.hero(
                extract(prompt, 'title') || extract(prompt, 'headline') || 'Your Page Title',
                extract(prompt, 'subtitle') || extract(prompt, 'headline') || 'Your subtitle goes here',
                extract(prompt, 'cta') || 'Learn More'
            );
        }

        function extract(prompt, type) {
            const patterns = {
                title: /(?:title|name)[:\s]+["']?([^"'\n]+)/i,
                subtitle: /subtitle[:\s]+["']?([^"'\n]+)/i,
                cta: /(?:cta|button|action)[:\s]+["']?([^"'\n]+)/i
            };
            const match = prompt.match(patterns[type]);
            return match ? match[1].trim() : null;
        }
